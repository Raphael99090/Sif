-- Anti-Lag (Remove Texturas)
local function AtivarAntiLag()
	for _, v in pairs(game:GetDescendants()) do
		if v:IsA("Part") or v:IsA("MeshPart") then
			v.Material = Enum.Material.SmoothPlastic
		elseif v:IsA("Texture") or v:IsA("Decal") then
			v:Destroy()
		end
	end
	Library:Notificar("Anti-Lag", "Texturas removidas (FPS Boost)", 3)
end

-- Auto Equip (Pega item na mÃ£o)
local function EquiparFerramenta()
	if not Config.AutoEquip then return end
	local Backpack = Player:FindFirstChild("Backpack")
	local Char = Player.Character
	if Backpack and Char then
		local Tool = Backpack:FindFirstChildOfClass("Tool")
		if Tool then Char.Humanoid:EquipTool(Tool) end
	end
end

-- FunÃ§Ãµes de Texto e Tempo
local function ConverterParaExtensoCaps(n)
	n = math.floor(tonumber(n) or 0)
	if n == 0 then return "ZERO" end
	if n == 100 then return "CEM" end
	if n > 999 then return tostring(n) end
	local U = {"", "UM", "DOIS", "TRÃŠS", "QUATRO", "CINCO", "SEIS", "SETE", "OITO", "NOVE"}
	local E = {"DEZ", "ONZE", "DOZE", "TREZE", "QUATORZE", "QUINZE", "DEZESSEIS", "DEZESSETE", "DEZOITO", "DEZENOVE"}
	local D = {"", "", "VINTE", "TRINTA", "QUARENTA", "CINQUENTA", "SESSENTA", "SETENTA", "OITENTA", "NOVENTA"}
	local C = {"", "CENTO", "DUZENTOS", "TREZENTOS", "QUATROCENTOS", "QUINHENTOS", "SEISCENTOS", "SETECENTOS", "OITOCENTOS", "NOVECENTOS"}
	local res = ""
	if n >= 100 then local c = math.floor(n/100); res = C[c+1]; n=n%100; if n>0 then res=res.." E " end end
	if n >= 10 and n <= 19 then res = res .. E[n-9]
	elseif n >= 20 then local d = math.floor(n/10); res = res .. D[d+1]; local u = n%10; if u>0 then res = res.." E "..U[u+1] end
	elseif n > 0 then res = res .. U[n+1] end
	return res
end

local function FormatarTempo(segundos)
	if segundos < 0 then segundos = 0 end
	return string.format("%02d:%02d", math.floor(segundos/60), math.floor(segundos%60))
end

local function ExecutarFisica()
	if not Estado.ScriptVivo then return end
	local char = Player.Character
	local hum = char and char:FindFirstChild("Humanoid")
	local root = char and char:FindFirstChild("HumanoidRootPart")
	
	if Config.Modo == "FlexÃ£o" then return end

	-- Auto Equipar antes de fazer o exercÃ­cio
	EquiparFerramenta()

	if Config.AutoAgachar then
		VirtualInputManager:SendKeyEvent(true, Enum.KeyCode.C, false, game)
		task.wait()
		VirtualInputManager:SendKeyEvent(false, Enum.KeyCode.C, false, game)
		task.wait(0.15)
		VirtualInputManager:SendKeyEvent(true, Enum.KeyCode.C, false, game)
		task.wait()
		VirtualInputManager:SendKeyEvent(false, Enum.KeyCode.C, false, game)
	end
	
	if hum then hum.Jump = true end
	
	if root and hum then
		task.wait(0.05)
		hum.AutoRotate = false
		task.spawn(function()
			local alvo = 360 + math.random(-20, 20)
			local girado = 0
			while girado < alvo and Estado.ScriptVivo do
				local passo = 50
				root.CFrame = root.CFrame * CFrame.Angles(0, math.rad(-passo), 0)
				girado = girado + passo
				task.wait(0.015)
			end
			hum.AutoRotate = true
		end)
	end
end

local AtualizarBotao
AtualizarBotao = AbaHome:CriarBotao("INICIAR TREINO", function()
	if Estado.Rodando then
		Estado.Rodando = false
		AtualizarBotao("INICIAR TREINO")
		DisplayETA("TEMPO: PARADO")
		Library:Notificar("Parado", "Treino pausado.", 2)
		return
	end

	Estado.Rodando = true
	AtualizarBotao("PARAR TREINO")
	Library:Notificar("Iniciado", "Contando...", 3)

	task.spawn(function()
		local fim = Config.Inicio + Config.Qtd - 1
		for i = Config.Inicio, fim do
			if not Estado.Rodando or not Estado.ScriptVivo then break end

			DisplayContador(tostring(i))
			local restante = (fim - i) * Config.Delay
			DisplayETA("RESTANTE: " .. FormatarTempo(restante))

			local msg = ConverterParaExtensoCaps(i) .. " !"
			if TextChatService.ChatInputBarConfiguration.TargetTextChannel then
				TextChatService.ChatInputBarConfiguration.TargetTextChannel:SendAsync(msg)
			else
				game:GetService("ReplicatedStorage").DefaultChatSystemChatEvents.SayMessageRequest:FireServer(msg, "All")
			end

			ExecutarFisica()

			local tempoEspera = Config.Delay
			if Config.Modo ~= "FlexÃ£o" then tempoEspera = tempoEspera - 0.3 end
			if tempoEspera < 0 then tempoEspera = 0 end
			task.wait(tempoEspera)
		end

		Estado.Rodando = false
		if Estado.ScriptVivo then
			AtualizarBotao("INICIAR TREINO")
			DisplayContador("FIM")
			DisplayETA("CONCLUÃDO")
			Library:Notificar("Sucesso", "Treino finalizado!", 5)
		end
	end)
end)
